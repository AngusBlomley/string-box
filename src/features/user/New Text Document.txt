import { createSlice } from '@reduxjs/toolkit';

export const userSlice = createSlice({
    name: 'user',
    initialState: {
        isAuthenticated: false,
        // other initial state
    },
    reducers: {
        logIn: (state, action) => {
            state.isAuthenticated = true;
            // other state updates
        },
        // other reducers
    },
});

export const { logIn } = userSlice.actions;

export default userSlice.reducer;
